import numpy as np
import matplotlib.pyplot as plt

def rosenbrock(x, y):
    return (1 - x)**2 + 100 * (y - x**2)**2

x = np.arange(-2, 2, 0.01)
y = np.arange(-1, 3, 0.01)
X, Y = np.meshgrid(x, y)
Z = rosenbrock(X, Y)

plt.contour(X, Y, Z, 50, cmap='jet')
plt.xlabel('x')
plt.ylabel('y')
plt.title('Rosenbrock Function Contour Plot')
plt.show()
